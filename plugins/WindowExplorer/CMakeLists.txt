    "prpsh.h"
    "resource.h"
    "wndexp.h"
    "wndtree.h"
)

    "version.rc"
    "WindowExplorer.rc"
)
source_group("Resource Files" FILES ${RESOURCES})

    "main.c"
    "prpsh.c"
    "utils.c"
    "wnddlg.c"
    "wndprp.c"
    "wndtree.c"
)

    /DELAYLOAD:comctl32.dll
    /DELAYLOAD:gdi32.dll
    /DELAYLOAD:ole32.dll
    /DELAYLOAD:propsys.dll
    /DELAYLOAD:shell32.dll
    /DELAYLOAD:user32.dll
)
